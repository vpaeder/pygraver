40 template<typename T> typename std::enable_if<!std::numeric_limits<T>::is_integer, bool>::type almost_equal(const T & x, const T & y, const int ulp) {
41 // the machine epsilon has to be scaled to the magnitude of the values used
42 // and multiplied by the desired precision in ULPs (units in the last place)
43 return std::abs(x-y) <= std::numeric_limits<T>::epsilon() * std::abs(x+y) * ulp * 1e2
44 // unless the result is subnormal
45 || std::abs(x-y) < std::numeric_limits<T>::min();
46 }

169 static std::tuple<size_t, size_t, size_t> convert_slice(const pybind11::slice & slice, const size_t arr_sz) {
170 ssize_t start, stop, step;
171 PySlice_Unpack(slice.ptr(), &start, &stop, &step);
172 if (start<0)
173 start = arr_sz - start;
174 if (stop<0)
175 stop = arr_sz - stop;
176 if (start > stop && step < 0)
177 return {stop, start, -step};
178 if (stop > start && step > 0)
179 return {start, stop, step};
180
181 PyErr_SetString(PyExc_ValueError, "Invalid slice.");
182 throw py::error_already_set();
183 return {0,0,0};

40 template<typename T> typename std::enable_if<!std::numeric_limits<T>::is_integer, bool>::type almost_equal(const T & x, const T & y, const int ulp) {
41 // the machine epsilon has to be scaled to the magnitude of the values used
42 // and multiplied by the desired precision in ULPs (units in the last place)
43 return std::abs(x-y) <= std::numeric_limits<T>::epsilon() * std::abs(x+y) * ulp * 1e2
44 // unless the result is subnormal
45 || std::abs(x-y) < std::numeric_limits<T>::min();
46 }

154 static std::tuple<size_t, size_t, size_t> convert_slice(const pybind11::slice & slice, const size_t arr_sz) {
155 ssize_t start, stop, step;
156 PySlice_Unpack(slice.ptr(), &start, &stop, &step);
157 if (start<0)
158 start = arr_sz - start;
159 if (stop<0)
160 stop = arr_sz - stop;
161 if (start > stop && step < 0)
162 return {stop, start, -step};
163 if (stop > start && step > 0)
164 return {start, stop, step};
165
166 PyErr_SetString(PyExc_ValueError, "Invalid slice.");
167 throw py::error_already_set();
168 return {0,0,0};
169 }

54 void from_string(const std::string & curve_str) {
55 std::string obj_str = curve_str;
56 // curve in SVG format (e.g. "M 10,10 L 5,0 ..." or "M10,10L5,0" or "M10 10L3 5")
57 std::regex segment_regex("[MmLlHhVvCcSsQqTtAaZz] *([-0123456789eE.]+(,| |)+)*");
58 std::regex args_regex("[-0123456789eE.]+");
59 // implicit flag; SVG commands can be implicitly specified (e.g. "m 10,10 5,0" ~ "m 10,10 m 5,0");
60 // if we find numbers at the end of a command instead of another command,
61 // we try to treat these numbers as a new command of the same type
62 std::string current_command = "", previous_command = "";
63 // start and end points for current curve segment
64 std::vector<T> p0 = {0,0}, p1 = {0,0}, p2 = {0,0}, p3 = {0,0};
65 PYG_LOG_V("Constructing curve...");
66 for (auto seg_it = std::sregex_iterator(obj_str.begin(), obj_str.end(), segment_regex); seg_it != std::sregex_iterator(); ++seg_it) {
67 std::string seg_match = seg_it->str();
68 current_command = seg_match[0];
69
70 for (auto arg_it = std::sregex_iterator(seg_match.begin()+1, seg_match.end(), args_regex); arg_it != std::sregex_iterator(); ++arg_it) {
71 if (current_command == "M" || current_command == "m") {
72 // move to - absolute
73 p1[0] = parse_value<T>((arg_it++)->str());
74 p1[1] = parse_value<T>(arg_it->str());
75 if (current_command == "m")
76 vec2_add<T>(p1, p0);
77 PYG_LOG_V("Found move-to command with parameters:");
78 PYG_LOG_V(" destination: ({},{})", p1[0], p1[1]);
79 // in case there's an implicit command, next point should be considered as a line
80 current_command = (current_command == "M") ? "L" : "l";
81 } else if (current_command == "L" || current_command == "l") {
82 // line
83 p1[0] = parse_value<T>((arg_it++)->str());
84 p1[1] = parse_value<T>(arg_it->str());
85 if (current_command == "l")
86 vec2_add<T>(p1, p0);
87 PYG_LOG_V("Found line with parameters:");
88 PYG_LOG_V(" end point: ({},{})", p1[0], p1[1]);
89 this->segments.push_back(std::make_unique<Line<T>>(p0, p1));
90 } else if (current_command == "H" || current_command == "h") {
91 // horizontal line
92 p1[0] = parse_value<T>(arg_it->str());
93 p1[1] = p0[1];
94 if (current_command == "h")
95 p1[0] += p0[0];
96 PYG_LOG_V("Found horizontal line with parameters:");
97 PYG_LOG_V(" end point: ({},{})", p1[0], p1[1]);
98 this->segments.push_back(std::make_unique<Line<T>>(p0, p1));
99 } else if (current_command == "V" || current_command == "v") {
100 // vertical line - absolute
101 p1[0] = p0[0];
102 p1[1] = parse_value<T>(arg_it->str());
103 if (current_command == "v")
104 p1[1] += p0[1];
105 PYG_LOG_V("Found vertical line with parameters:");
106 PYG_LOG_V(" end point: ({},{})", p1[0], p1[1]);
107 this->segments.push_back(std::make_unique<Line<T>>(p0, p1));
108 } else if (current_command == "C" || current_command == "c") {
109 // cubic bezier
110 p2[0] = parse_value<T>((arg_it++)->str());
111 p2[1] = parse_value<T>((arg_it++)->str());
112 p3[0] = parse_value<T>((arg_it++)->str());
113 p3[1] = parse_value<T>((arg_it++)->str());
114 p1[0] = parse_value<T>((arg_it++)->str());
115 p1[1] = parse_value<T>(arg_it->str());
116 if (current_command == "c") {
117 vec2_add<T>(p1, p0);
118 vec2_add<T>(p2, p0);
119 vec2_add<T>(p3, p0);
120 }
121 PYG_LOG_V("Found cubic bezier curve with parameters:");
122 PYG_LOG_V(" handle point 1: ({},{})", p2[0], p2[1]);
123 PYG_LOG_V(" handle point 2: ({},{})", p3[0], p3[1]);
124 PYG_LOG_V(" end point: ({},{})", p1[0], p1[1]);
125 this->segments.push_back(std::make_unique<Bezier3<T>>(p0, p2, p3, p1));
126 } else if (current_command == "S" || current_command == "s") {
127 // smooth cubic bézier
128 if (previous_command == "C" || previous_command == "S"
129 || previous_command == "c" || previous_command == "s") {
130 p2[0] = p0[0] - p3[0];
131 p2[1] = p0[1] - p3[1];
132 } else {
133 p2[0] = p0[0];
134 p2[1] = p0[1];
135 }
136 p3[0] = parse_value<T>((arg_it++)->str());
137 p3[1] = parse_value<T>((arg_it++)->str());
138 p1[0] = parse_value<T>((arg_it++)->str());
139 p1[1] = parse_value<T>(arg_it->str());
140 if (current_command == "s") {
141 vec2_add<T>(p1, p0);
142 vec2_add<T>(p3, p0);
143 }
144 PYG_LOG_V("Found smooth cubic bezier curve with parameters:");
145 PYG_LOG_V(" handle point 1: ({},{})", p2[0], p2[1]);
146 PYG_LOG_V(" handle point 2: ({},{})", p3[0], p3[1]);
147 PYG_LOG_V(" end point: ({},{})", p1[0], p1[1]);
148 this->segments.push_back(std::make_unique<Bezier3<T>>(p0, p2, p3, p1));
149 } else if (current_command == "Q" || current_command == "q") {
150 // quadratic bezier
151 p2[0] = parse_value<T>((arg_it++)->str());
152 p2[1] = parse_value<T>((arg_it++)->str());
153 p1[0] = parse_value<T>((arg_it++)->str());
154 p1[1] = parse_value<T>(arg_it->str());
155 if (current_command == "q") {
156 vec2_add<T>(p2, p0);
157 vec2_add<T>(p1, p0);
158 }
159 PYG_LOG_V("Found quadratic bezier curve with parameters:");
160 PYG_LOG_V(" handle point 1: ({},{})", p2[0], p2[1]);
161 PYG_LOG_V(" end point: ({},{})", p1[0], p1[1]);
162 this->segments.push_back(std::make_unique<Bezier3<T>>(p0, p2, std::vector<T>{0,0}, p1));
163 } else if (current_command == "T" || current_command == "t") {
164 // smooth quadratic bézier
165 if (previous_command == "T" || previous_command == "Q"
166 || previous_command == "t" || previous_command == "q") {
167 p2[0] = p1[0] - p2[0];
168 p2[1] = p1[1] - p2[1];
169 } else {
170 p2[0] = p0[0];
171 p2[1] = p0[1];
172 }
173 p1[0] = parse_value<T>((arg_it++)->str());
174 p1[1] = parse_value<T>(arg_it->str());
175 if (current_command == "t")
176 vec2_add<T>(p1, p0);
177 PYG_LOG_V("Found smooth quadratic bezier curve with parameters:");
178 PYG_LOG_V(" handle point 1: ({},{})", p2[0], p2[1]);
179 PYG_LOG_V(" end point: ({},{})", p1[0], p1[1]);
180 this->segments.push_back(std::make_unique<Bezier3<T>>(p0, p2, std::vector<T>{0,0}, p1));
181 } else if (current_command == "A" || current_command == "a") {
182 // arc
183 std::vector<T> r(2);
184 r[0] = parse_value<T>((arg_it++)->str());
185 r[1] = parse_value<T>(arg_it->str());
186 // the SVG TR specifies a number of invalid cases and how to handle them;
187 // e.g. if only one of the radii is 0, we generate a line instead;
188 // I assume SVG generators already handle that, so if one radius is 0 I just
189 // dump the segment (check your SVG);
190 // the other case is when p0 and p1 are specified with incompatible radii;
191 // in this case, the arc code tries to find a suitable combination
192 // that permits connecting p0 and p1 with an arc, and throws an exception
193 // if it fails; in this case, debug logs won't reflect adjustments.
194 if (r[0]!=0 && r[1]!=0) {
195 arg_it++;
196 T angle = parse_value<T>((arg_it++)->str())/180*M_PI;
197 int large_arc_flag = std::stoi((arg_it++)->str());
198 int sweep_flag = std::stoi((arg_it++)->str());
199 p1[0] = parse_value<T>((arg_it++)->str());
200 p1[1] = parse_value<T>(arg_it->str());
201 if (current_command == "a")
202 vec2_add<T>(p1, p0);
203 PYG_LOG_V("Found arc with parameters:");
204 PYG_LOG_V(" radii: ({},{})", r[0], r[1]);
205 PYG_LOG_V(" tilt angle: {} degrees", angle/M_PI*180);
206 PYG_LOG_V(" large arc flag is {}set", large_arc_flag ? "" : "not ");
207 PYG_LOG_V(" sweep flag is {}set", sweep_flag ? "" : "not ");
208 PYG_LOG_V(" end point: ({},{})", p1[0], p1[1]);
209 this->segments.push_back(std::make_unique<Arc<T>>(p0, p1, r, angle, large_arc_flag==1, sweep_flag==1));
210 }
211 } else {
212 PYG_LOG_I("Unknown SVG curve segment type: {}", current_command);
213 }
214 previous_command = current_command;
215 p0 = p1;
216 PYG_LOG_V("Next segment starting at ({},{})", p0[0], p0[1]);
217 }
218 }
219 if (current_command == "z" || current_command == "Z") {
220 // close path
221 p0 = this->segments.back()->point(1);
222 p1 = this->segments[0]->point(0);
223 PYG_LOG_V("Found closing line. Adding line from ({},{}) to ({},{})", p0[0], p1[1], p1[0], p1[1]);
224 this->segments.push_back(std::make_unique<Line<T>>(p0, p1));
225 this->is_closed = true;
226 }
227 }

233 std::vector<std::vector<T>> interpolate(const T dl) const override {
234 PYG_LOG_D("Interpolating curve 0x{:x}", (uint64_t)this);
235 PYG_LOG_D("Number of segments: {}", this->segments.size());
236 std::vector<std::vector<T>> points;
237 for (auto & seg : this->segments) {
238 auto new_points = seg->interpolate(dl);
239 points.reserve(points.size() + new_points.size());
240 points.insert(points.end(), new_points.begin(), new_points.end());
241 }
242 if (!this->is_closed)
243 points.push_back(this->segments.back()->point(1));
244
245 return points;
246 }

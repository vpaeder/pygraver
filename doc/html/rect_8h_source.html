57 bool has_rx = false, has_ry = false;
58 for (auto attr = node->properties; attr; attr = attr->next) {
59 if (!xmlStrcmp(attr->name, (const xmlChar *)"rx")) {
60 rx = get_number<T>(node, "rx", w);
61 has_rx = true;

63 ry = get_number<T>(node, "ry", h);
64 has_ry = true;
65 }
66 }
67 if (has_rx && !has_ry)

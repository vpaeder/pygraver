32 template <typename T> static T carlson_rf(const T x, const T y, const T z, const T errtol) {
33 if (x<0 || y<0 || z<0)
34 throw std::invalid_argument{"x, y, and z must be positive"};
35
36 T eps, mu;
37 T xn = x, yn = std::max(y, std::numeric_limits<T>::min()), zn = z;
38 T xndev, yndev, zndev;
39 T xnroot, ynroot, znroot;
40 T lambda;
41
42 unsigned int iter = ELLIPTIC_MAX_ITER;
43 do {
44 mu = (xn + yn + zn)/3;
45 xndev = 2 - (mu + xn)/mu;
46 yndev = 2 - (mu + yn)/mu;
47 zndev = 2 - (mu + zn)/mu;
48 eps = std::max({xndev, -xndev, yndev, -yndev, zndev, -zndev});
49 if (eps < errtol) break;
50 xnroot = sqrt(xn);
51 ynroot = sqrt(yn);
52 znroot = sqrt(zn);
53 lambda = xnroot * (ynroot + znroot) + ynroot * znroot;
54 xn = (xn + lambda)/4;
55 yn = (yn + lambda)/4;
56 zn = (zn + lambda)/4;
57 } while (--iter);
58
59 T e1 = xndev * yndev;
60 T e2 = e1 - zndev * zndev;
61 T e3 = e1 * zndev;
62 T s = 1 + (e2/24 - 0.1 - 3*e3/44) * e2 + e3/14;
63
64 return s / sqrt(mu);
65 }

75 template <typename T> static T carlson_rd(const T x, const T y, const T z, const T errtol) {
76 if (x<0 || y<0 || z<0)
77 throw std::invalid_argument{"x, y, and z must be positive"};
78
79 T eps, mu;
80 T xn = x, yn = std::max(y, std::numeric_limits<T>::epsilon()), zn = z;
81 T pow4 = 1;
82 T xndev, yndev, zndev;
83 T xnroot, ynroot, znroot;
84 T lambda, sigma=0;
85
86 unsigned int iter = ELLIPTIC_MAX_ITER;
87 do {
88 mu = (xn + yn + 3 * zn)/5;
89 xndev = (mu - xn) / mu;
90 yndev = (mu - yn) / mu;
91 zndev = (mu - zn) / mu;
92 eps = std::max({xndev, -xndev, yndev, -yndev, zndev, -zndev});
93 if (eps < errtol) break;
94 xnroot = sqrt(xn);
95 ynroot = sqrt(yn);
96 znroot = sqrt(zn);
97 lambda = xnroot * (ynroot + znroot) + ynroot * znroot;
98 sigma += pow4 / (znroot * (zn + lambda));
99 pow4 /= 4;
100 xn = (xn + lambda)/4;
101 yn = (yn + lambda)/4;
102 zn = (zn + lambda)/4;
103 } while (--iter);
104
105 T ea = xndev * yndev;
106 T eb = zndev * zndev;
107 T ec = ea - eb;
108 T ed = ea - 6*eb;
109 T ef = ed + ec + ec;
110 T s1 = ed * (9/88*ed - 9/52*zndev * ef - 3/14);
111 T s2 = zndev * (ef/6 + zndev * (zndev * ea*3/26 - ec*9/22));
112 return 3 * sigma + pow4 * (1 + s1 + s2) / (mu * sqrt(mu));
113 }

121 template <typename T> static T elliptic_e(const T k, const T errtol) {
122 T kk = k*k;
123 return carlson_rf<T>(0, 1-kk, 1, errtol) - kk/3*carlson_rd<T>(0, 1-kk, 1, errtol);

133 template <typename T> static T elliptic_e(const T phi, const T k, const T errtol) {
134 // The original version works for phi in [-pi/2,pi/2] (+/- 1/4 of an ellipse).
135 // Because the integrand is periodic, we can derive values beyond pi/2
136 // by adding quarter ellipse lengths until reaching the right quadrant.
137 int mf = (phi>=0) ? (phi + M_PI_2)/M_PI : (phi - M_PI_2)/M_PI;
138 T c = cos(phi), s = sin(phi);
139 T x = c*c, ss = s*s, kk = k*k;
140 T y = 1 - kk*ss;
141 T v = s*(carlson_rf<T>(x, y, 1, errtol) - kk*s*s/3*carlson_rd<T>(x, y, 1, errtol));
142 T corr = 0;
143 if (mf)
144 corr = 2*mf*elliptic_e(k, errtol);
145 if (mf & 1)
146 return corr - v;
147 else
148 return corr + v;
149 }

164 template <typename T> static T inv_elliptic_e(const T L, const T k, const T errtol) {
165 T E1 = elliptic_e(k, errtol);
166 T zeta = 1 - L/E1;
167 T mu = 1 - k;
168 T r = sqrt(zeta * zeta + mu * mu);
169 T theta = atan(mu/(L + std::numeric_limits<T>::epsilon()));
170 T res = M_PI_2 + sqrt(r) * (theta - M_PI_2);
171 T nres;
172 T s;
173 unsigned int iter = ELLIPTIC_MAX_ITER;
174 do {
175 s = sin(res);
176 nres = res - (elliptic_e(res, k, errtol) - L)/sqrt(1 - k*s*s);
177 if (abs(nres - res) < errtol) break;
178 res = nres;
179 } while (--iter);
180 return nres;
181 }

251 Arc(const std::vector<T> & start, const std::vector<T> & end,
252 const std::vector<T> & r, const T angle, const bool large_arc_flag,
253 const bool sweep_flag) {
254 this->p0 = start;
255 this->r = r;
256 this->ca = cos(angle);
257 this->sa = sin(angle);
258 this->is_circle = r[0]==r[1];
259 // computation of arc angles following https://www.w3.org/TR/SVG/implnote.html#ArcImplementationNotes
260 std::vector<T> dp(2), p1(2), pc1(2), u(2), v(2);
261 dp[0] = (start[0] - end[0])/2.0;
262 dp[1] = (start[1] - end[1])/2.0;
263 //p1[0] = this->ca*dp[0] + this->sa*dp[1];
264 //p1[1] = -this->sa*dp[0] + this->ca*dp[1];
265 p1[0] = dp[0];
266 p1[1] = dp[1];
267 std::vector<T> p12 = vec2_pow<T>(p1, 2);
268 std::vector<T> r2 = vec2_pow<T>(r, 2);
269 T sign = (large_arc_flag ^ sweep_flag) ? 1 : -1;
270 // compute half-distance vector
271 T pc0_num = r2[0]*r2[1] - r2[0]*p12[1] - r2[1]*p12[0];
272 // if we cannot connect points with arc of given radius, pc0_num may be negative;
273 // in this case, we attempt to correct the problem by adjusting radii until pc0_num==0
274 while (pc0_num<0) {
275 r2[0] -= pc0_num/2;
276 r2[1] -= pc0_num/2;
277 if (r2[0]<0 || r2[1]<0)
278 throw std::runtime_error("SVG: cannot find suitable arc radii to connect endpoints.");
279 pc0_num = r2[0]*r2[1] - r2[0]*p12[1] - r2[1]*p12[0];
280 this->r[0] = sqrt(r2[0]);
281 this->r[1] = sqrt(r2[1]);
282 }
283 T pc0 = sign*sqrt(pc0_num/(r2[0]*p12[1]+r2[1]*p12[0]));
284 pc1[0] = pc0*r[0]*p1[1]/r[1];
285 pc1[1] = -pc0*r[1]*p1[0]/r[0];
286 // centre point
287 this->pc[0] = this->ca*pc1[0] - this->sa*pc1[1] + (start[0] + end[0])/2;
288 this->pc[1] = this->sa*pc1[0] + this->ca*pc1[1] + (start[1] + end[1])/2;
289 // start and end angles
290 u[0] = (p1[0]-pc1[0])/r[0];
291 u[1] = (p1[1]-pc1[1])/r[1];
292 v[0] = -(p1[0]+pc1[0])/r[0];
293 v[1] = -(p1[1]+pc1[1])/r[1];
294 T lu = sqrt(pow(u[0],2) + pow(u[1],2));
295 T lv = sqrt(pow(v[0],2) + pow(v[1],2));
296 this->t_start = acos(u[0]/lu)*sgn<T>(u[1]);
297 T dtheta = acos((u[0]*v[0] + u[1]*v[1])/lu/lv)*sgn<T>(u[0]*v[1]-u[1]*v[0]);
298 if (sweep_flag && dtheta<0) {
299 dtheta += M_2PI;
300 } else if (!sweep_flag && dtheta>0) {
301 dtheta -= M_2PI;
302 }
303 this->t_end = this->t_start + dtheta;
304 // store full length
305 this->_length = this->length(1);
306 }

363 return t*abs(this->t_end - this->t_start)*this->r[0];
